WebAPI-DOM尺寸和位置

让便签可被拖动，但不能超出视口​

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>可拖动的便笺</title>
    <link rel="stylesheet" href="./css/index.css" />
  </head>
  <body>
    <div class="note">
      <div class="move-bar"></div>
      <div class="content" contenteditable="">
        <p>这是一个便笺</p>
        <p>里面的文字可以更改</p>
        <p>用鼠标按住顶部的移动条即可拖动便笺</p>
      </div>
    </div>

    <script src="./js/index.js"></script>
  </body>
</html>

// 让便签可被拖动，但不能超出视口
let move = document.querySelector('.move-bar');
let note = document.querySelector('.note');

let rect = move.getBoundingClientRect();
let rectX = rect.left;
let rectY = rect.top;

console.log(rect, rectX, rectY);

let ex = document.documentElement.clientWidth;
let eh = document.documentElement.clientHeight;

let maxLeft = ex - rectX;
let maxTop = ex - rectY;

move.onmousedown = function () {
  window.onmousemove = function (e) {
    console.log(e.clientX, e.clientY);

    let left = e.clientX - rectX;
    let top = e.clientY - rectY;

    left < 0 ? left = 0 : (left > maxLeft ? left = maxLeft : left)
    top < 0 ? top = 0 : (top > maxTop ? top = maxTop : top)


    note.style.left = left + 'px';
    note.style.top = top + 'px';
  };

  window.onmouseup = function (e) {
    window.onmousemove = null;
    window.onmouseup = null;
  };
};

让小球向右下运动，遇到边缘后反弹​

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>自行运动的小球</title>
    <style>
      .ball {
        width: 50px;
        height: 50px;
        background: #f40;
        border-radius: 50%;
        position: fixed;
        left: 100px;
        top: 50px;
      }
    </style>
  </head>
  <body>
    <div class="ball"></div>

    <script src="./js/index.js"></script>
  </body>
</html>

// 让小球向右下运动，遇到边缘后反弹

let ball = document.querySelector('.ball');

let disX = 1;
let disY = 1;

let w = document.documentElement.clientWidth;
let h = document.documentElement.clientHeight;
let ew = ball.clientWidth;
let eh = ball.clientHeight;

let mw = w - ew;
let mh = h - eh;

function getRandom(max, min) {
  return Math.floor(Math.random() * (max - min) + min);
}

function randomBG() {
  let r = getRandom(0, 256);
  let g = getRandom(0, 256);
  let b = getRandom(0, 256);
  ball.style.background = `rgb(${r} ,${g} ,${b})`;
}

setInterval(function () {
  let rect = ball.getBoundingClientRect();
  let x = rect.left;
  let y = rect.top;

  let left = x + disX;
  let top = y + disY;

  if (left < 0) {
    left = 0;
    disX = -disX;
    randomBG();
  }
  if (top < 0) {
    top = 0;
    disY = -disY;
    randomBG();
  }
  if (left > mw) {
    left = mw;
    disX = -disX;
    randomBG();
  }
  if (top > mh) {
    top = mh;
    disY = -disY;
    randomBG();
  }

  // Correct the style setting
  ball.style.left = `${left}px`;
  ball.style.top = `${top}px`;
}, 2);
