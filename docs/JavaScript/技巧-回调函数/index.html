function filter(callback) {
  let result = [];
  students.forEach((item) => {
    if (callback(item)) {
      result.push(item);
    }
  });
  return result;
}

// 得到所有女生
let a = filter((s) => {
  return s.sex === '女';
});

// 得到所有20岁以下的女生
let b = filter((s) => {
  return s.age < 20;
});

// 得到所有姓赵的同学

let c = filter((s)=>{
  return s.name.substring(0,1) === '赵'
})


// 得到所有居住在黑龙江的女生
let d = filter((s)=>{
  return s.address.indexOf('黑龙江') !== -1
